🚀 HOW TO USE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. IMPORT
     ────────────────────────────────────────────────────────────────────────
     import TournamentBracket, { BracketSkeleton } 
       from '...TournamentBracket';


  2. PREPARE DATA
     ────────────────────────────────────────────────────────────────────────
     const matches = [
       {
         _id: "m1",
         matchNumber: 1,
         round: 1,
         status: "completed",
         participant1: "Team A",
         participant1Uucms: "USMS123",
         participant2: "Team B",
         participant2Uucms: "USMS456",
         score1: 3,
         score2: 1,
         winner: "Team A",
         scheduledTime: "2024-03-31T14:00:00Z"
       },
       // ... more matches ...
     ];

     const totalRounds = matches.length > 0
       ? Math.max(...matches.map(m => m.round))
       : 0;


  3. RENDER
     ────────────────────────────────────────────────────────────────────────
     <div>
       {loading ? (
         <BracketSkeleton />
       ) : (
         <TournamentBracket 
           matches={matches} 
           totalRounds={totalRounds} 
         />
       )}
     </div>


  4. THAT'S IT! 🎉
     ────────────────────────────────────────────────────────────────────────
     Component handles everything:
     • Display all rounds horizontally
     • Show match details on click
     • Animate winner changes
     • Update in real-time via Socket.io
     • Responsive on all devices
